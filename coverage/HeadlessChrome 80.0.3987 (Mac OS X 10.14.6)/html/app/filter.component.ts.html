import {Component, Input} from "@angular/core";
 
export type Filter = {
    query?: string;
    sortBy: string;
    sortDir: "asc" | "desc";
}
 
@Component({
    selector: "filter",
    template: `
        <div class="filter">
            <div class="sort">
                <button (click)="toggleSort('name')" [ngClass]="{active: filter.sortBy == 'name'}">City <i
                        [ngClass]="['sort-alpha', nextDir('name')]"></i></button>
                <button (click)="toggleSort('main.temp')" [ngClass]="{active: filter.sortBy == 'main.temp'}">Temp. <i
                        [ngClass]="['sort-numeric', nextDir('main.temp')]"></i></button>
            </div>
            <div class="search">
                <label>
                    <svg height="24" viewBox="0 0 24 24" width="24"><path d="M15.5 14h-.79l-.28-.27C15.41 12.59 16 11.11 16 9.5 16 5.91 13.09 3 9.5 3S3 5.91 3 9.5 5.91 16 9.5 16c1.61 0 3.09-.59 4.23-1.57l.27.28v.79l5 4.99L20.49 19l-4.99-5zm-6 0C7.01 14 5 11.99 5 9.5S7.01 5 9.5 5 14 7.01 14 9.5 11.99 14 9.5 14z"/><path d="M0 0h24v24H0z" fill="none"/></svg>
                    <input type="text" [(ngModel)]="filter.query">
                </label>
            </div>
        </div>`,
    styleUrls: ["filter.component.scss"]
})
export class FilterComponent {
    @Input()
    public filter: Filter;
 
    public nextDir = (param: string) => this.filter.sortBy === param && this.filter.sortDir === "asc" ? "desc" : "asc";
 
    public toggleSort(param: string) {
        if(this.filter.sortBy === param) {
            this.filter.sortDir = this.filter.sortDir === "asc" ? "desc" : "asc";
        } else {
            this.filter.sortBy = param;
            this.filter.sortDir = "asc";
        }
    }
}
